    const formdata = new FormData();

    const url = `${process.env.APP_LINK}`;
    const token = localStorage.getItem('rest_token');
 
    axios.post(`${url}csv/products`, formdata, {
      headers: {
        Authorization: `Token ${token}`,
      }
    })
      .then((res) => {
        const { noOfProductsAdded } = res.data;
        this.setState({
          loading: false,
